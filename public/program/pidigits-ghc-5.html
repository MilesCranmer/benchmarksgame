-- Pseudo FFI CTypes
type CWord = Word32
type CInt# = Int#
narrowCInt# :: Int# -> Int#
narrowCInt# = narrow32Int#

data MPZ = MPZ { alloc :: !Int32, size :: !Int32, ptr :: !IntPtr }

-- to make foreign pointers work...
instance Storable MPZ where
  sizeOf mpz = (sizeOf (0:: Int32)) * 2 + alignment mpz
  alignment _ = sizeOf (0 :: IntPtr)
  peek mpzp = do
    a <- peekElemOff (castPtr mpzp) 0
    s <- peekElemOff (castPtr mpzp) 1
    p <- peekByteOff (castPtr mpzp) ((sizeOf (0 :: Int32)) * 2)
    return $! MPZ a s p
  poke ptrmpz (MPZ a s p) = do
    _ <- pokeElemOff (castPtr ptrmpz) 0 a
    _ <- pokeElemOff (castPtr ptrmpz) 1 s
    _ <- pokeByteOff (castPtr ptrmpz) ((sizeOf (0 :: Int32)) * 2) p
    return ()

newtype MPZPtr = MPZPtr (Ptr MPZ)
 
instance Eq MPZPtr where
  (==) = eqMPZPtr

instance Ord MPZPtr where
  compare = compareMPZPtr

-- monadic combinators for low-level state threading
type S s a = State# s -> (# State# s, a #)

runS :: S RealWorld a -> a
runS m = case runRW# m of (# _, a #) -> a `seq` a

{-# INLINE liftIO #-}
liftIO :: IO a -> S RealWorld a
liftIO (IO m) = m

{-# INLINE wrapper2MPZUi #-}
wrapper2MPZUi :: (Ptr MPZ -> Ptr MPZ -> CWord -> IO ()) ->
                                        MPZPtr -> MPZPtr -> CWord -> MPZPtr
wrapper2MPZUi f (MPZPtr rtrnp) (MPZPtr passp) gl = runS $ \ s0 ->
  case liftIO (f rtrnp passp gl) s0 of
    (# s1, _ #) -> case liftIO (peek rtrnp) s1 of
      (# s2, _ #) -> (# s2, MPZPtr $! rtrnp #)

{-# INLINE eqMPZPtr #-}
eqMPZPtr :: MPZPtr -> MPZPtr -> Bool
eqMPZPtr mpzptra mpzptrb =
  case EQ == compareMPZPtr mpzptra mpzptrb of rslt -> rslt

{-# INLINE compareMPZPtr #-}
compareMPZPtr :: MPZPtr -> MPZPtr -> Ordering
compareMPZPtr (MPZPtr mpzpa) (MPZPtr mpzpb) =
  case compareInt# (narrowCInt# (mpzCmp mpzpa mpzpb)) 0# of r -> r

setMPZPtr :: Ptr MPZ -> CWord -> MPZPtr
setMPZPtr mpzptr gl = runS $ \ s0 ->
  case liftIO (mpzInitSetSi mpzptr gl) s0 of
    (# s1, _ #) -> case liftIO (peek mpzptr) s1 of
      (# s2, _ #) -> (# s2, MPZPtr mpzptr #)

{-# INLINE getMPZPtr #-}
getMPZPtr :: MPZPtr -> CWord
getMPZPtr (MPZPtr mpzp) = case mpzGetSi mpzp of rslt -> rslt

{-# INLINE addMPZPtr #-}
addMPZPtr :: MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr
addMPZPtr (MPZPtr sump) (MPZPtr frstp) (MPZPtr scndp) = runS $ \ s0 ->
  case liftIO (mpzAdd sump frstp scndp) s0 of
    (# s1, _ #) -> case liftIO (peek sump) s1 of
      (# s2, _ #) -> (# s2, MPZPtr sump #)

{-# INLINE divMPZPtr #-}
divMPZPtr :: MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr
divMPZPtr (MPZPtr quotp) (MPZPtr np) (MPZPtr dp) = runS $ \ s0 ->
  case liftIO (mpzTdivQ quotp np dp) s0 of
    (# s1, _ #) -> case liftIO (peek quotp) s1 of
      (# s2, _ #) -> (# s2, MPZPtr quotp #)

main :: IO ()
main = doit . read . head =<< getArgs

doit :: Int -> IO ()
doit numdgs =
    alloca $ \ accptr -> alloca $ \ denptr -> alloca $ \ numptr ->
      alloca $ \ tmp1ptr -> alloca $ \ tmp2ptr ->
        loop 1 9 $ digits 0 (setMPZPtr accptr 0)
                            (setMPZPtr denptr 1) (setMPZPtr numptr 1)
                            (setMPZPtr tmp1ptr 0) (setMPZPtr tmp2ptr 0)
  where loop i n ds =
          if i <= numdgs then do
            putStr $ show $ head ds
            if n <= 0 then putStrLn $ "\t:" ++ show i else return ()
            loop (i + 1) (if n <= 0 then 9 else n - 1) $! (tail ds)
          else do
            if n < 9 then putStrLn $ replicate n ' ' ++ "\t:" ++ show numdgs
            else return ()

-- list iterator of pi digits...
digits :: CWord -> MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr -> [CWord]
digits k acc den num tmp1 tmp2 =
  let nk = k + 1
      (nacc, nden, nnum) = nextTerm nk acc den num in
  if nnum > nacc then digits nk nacc nden nnum tmp1 tmp2 else

  let (d, tmp1a, tmp2a) = extractDigit 3 nacc nden nnum tmp1 tmp2 in
  let (t, tmp1b, tmp2b) =
        d `seq` tmp1a `seq` tmp2a `seq`
          extractDigit 4 nacc nden nnum tmp1a tmp2a in
  if d /= t then digits nk nacc nden nnum tmp1b tmp2b else

  let (nnacc, nnnum) = eliminateDigit d nacc nden nnum in
  d : (nnacc `seq` nnnum `seq` digits nk nnacc nden nnnum tmp1b tmp2b)

{-# INLINE nextTerm #-}
nextTerm :: CWord -> MPZPtr -> MPZPtr -> MPZPtr -> (MPZPtr, MPZPtr, MPZPtr)
nextTerm k acc den num = -- careful of side effects!...
  let k2 = k + k + 1
      nacc0 = wrapper2MPZUi mpzAddmulUi acc num 2
      nacc = nacc0 `seq` wrapper2MPZUi mpzMulUi nacc0 nacc0 k2 in
  let nden = wrapper2MPZUi mpzMulUi den den k2
      nnum = nacc `seq` wrapper2MPZUi mpzMulUi num num k
  in (nacc, nden, nnum)

{-# INLINE extractDigit #-}
extractDigit :: CWord -> MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr -> MPZPtr
                                              -> (CWord, MPZPtr, MPZPtr)
extractDigit nth acc den num tmp1 tmp2 = -- careful of side effects!...
  let tmp1a = wrapper2MPZUi mpzMulUi tmp1 num nth
      tmp2a = tmp1a `seq` addMPZPtr tmp2 tmp1a acc
      tmp1b = tmp2a `seq` divMPZPtr tmp1a tmp2a den
  in (getMPZPtr tmp1b, tmp1b, tmp2a)

{-# INLINE eliminateDigit #-}
eliminateDigit :: CWord -> MPZPtr -> MPZPtr -> MPZPtr -> (MPZPtr, MPZPtr)
eliminateDigit d acc den num =
  let acc0 = wrapper2MPZUi mpzSubmulUi acc den d
      nacc = wrapper2MPZUi mpzMulUi acc0 acc0 10
      nnum = wrapper2MPZUi mpzMulUi num num 10
  in (nacc, nnum)
