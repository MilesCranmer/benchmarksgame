const cp = require('child_process');
const rd = require('readline');

const prepareMap = (freq, length, prefix = '') => {
  ['a', 'c', 'g', 't'].forEach(item => {
    if (length > 1) {
      prepareMap(freq, length - 1, `${prefix}${item}`);
    } else {
      freq.set(`${prefix}${item}`, 0);
    }
  });

  return freq;
};

const frequency = (seq, s) => {
  let count = 0;
  let lastIndex = -1;
  let index;

  while ((index = seq.indexOf(s, lastIndex + 1)) > lastIndex) {
    ++count;
    lastIndex = index;
  }

  return count;
};

const sort = (seq, length) => {
  const freq = prepareMap(new Map(), length);
  const keys = [...freq.keys()];
  const n = seq.length - length + 1;

  keys.forEach(key => {
    freq.set(key, frequency(seq, key));
  });

  return `${
    keys
      .sort((a, b) => freq.get(b) - freq.get(a))
      .map(i => `${i.toUpperCase()} ${(freq.get(i) * 100 / n).toFixed(3)}`)
      .join('\n')
    }\n\n`;
};

const find = (seq, s) => `${frequency(seq, s)}\t${s.toUpperCase()}\n`;

if (process.env.i) {
  process.on('message', seq => {
    let res = '';

    switch (process.env.i) {
      case '1':
        res += sort(seq, 1);
        break;
      case '2':
        res += sort(seq, 2);
        break;
      case '3':
        res += find(seq, 'ggt');
        res += find(seq, 'ggta');
        res += find(seq, 'ggtatt');
        break;
      case '4':
        res += find(seq, 'ggtattttaatt');
        res += find(seq, 'ggtattttaatttatagt');
        break;
    }

    process.send(res);
    process.exit();
  });
} else {
  let reading = false;
  let seq = '';
  let jobs = 4;
  const result = [];

  rd.createInterface(process.stdin)
    .on('line', line => {
      if (reading) {
        seq += line;
      } else {
        reading = line.substr(0, 6) === '>THREE';
      }
    })
    .on('close', () => {
      for (let i = 1; i <= jobs; ++i) {
        cp.fork(__filename, [], { silent: true, env: { i } })
          .on('message', message => {
            result[i] = message;

            if (!(--jobs)) {
              process.stdout.write(result.join(''));
              process.exit(0);
            }
          })
          .send(seq);
      }
    });
}
