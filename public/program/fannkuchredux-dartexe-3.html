import 'dart:isolate';

const MAX_N = 16;

void fannkuch(Request r) {     
    final int n = r.n;    
    final List<int> Fact = r.Fact;     

    int max = r.min + r.CHUNKSZ;      
    if (max < Fact[n]) {   
        Isolate.spawn(fannkuch, Request(n,max,r.CHUNKSZ,Fact,r.ch) );       
    } else {        
        max = Fact[n];
    }
    
    // allocation
    var p = List<int>.filled(MAX_N,0),
        pp = List<int>.filled(MAX_N,0), 
        count = List<int>.filled(MAX_N,0);    
    
    // First permutation.
    for (int i = 0; i < n; i++) {
        p[i] = i;
    }    
    for (int i = n-1, idx = r.min; i > 0; i--) {
        final d = idx ~/ Fact[i];
        count[i] = d;
        idx = idx % Fact[i];
        
        List.copyRange(pp,0,p);
        for (int j = 0; j <= i; j++) {
            if (j+d <= i) {
                p[j] = pp[j+d];
            } else {
                p[j] = pp[j+d-i-1];
            }
        }        
    }    
       
    final reply = Reply(0,1);     

    for (var idx = r.min, sign = true; ; sign = !sign) { 
   
        // Count flips
        int first = p[0]; 
        if (first != 0) { 
            int flips = 1;
            if (p[first] != 0) {  
                List.copyRange(pp,0,p);         
                int p0 = first;
                for (;;) {
                    int t;                
                    flips++;                   
                    for (int i = 1, j = p0-1; i < j; i = i+1, j = j-1) {
                        t = pp[i]; pp[i] = pp[j]; pp[j] = t;
                    }   
                    t = pp[p0];
                    pp[p0] = p0; 
                    p0 = t;
                    if (pp[p0] == 0) {                  
                        break; 
                    }                    
                } 
            }   
            if (reply.flips < flips) {
                reply.flips = flips;
            }
            if (sign) {
                reply.sum += flips;
            } else {
                reply.sum -= flips;
            }  
        }             
                 
        idx++; if (idx == max) {                   
            break;
        }             
        
        // Next permutation
        if (sign) {
            p[0]= p[1]; p[1] = first;
        } else {
            int t = p[1]; p[1] = p[2]; p[2] = t;
            for (int k = 2; ; k++) {            
                count[k]++; if (count[k] <= k) {                                  
                    break;
                }                  
                count[k] = 0;
                for (int j = 0; j <= k; j++) {
                    p[j] = p[j+1];
                }
                p[k+1] = first;
                first = p[0];
            }
        }                              
    }     
    r.ch.send(reply);
}

void main(List<String> args) async {
    final n = (args.length > 0) ? int.parse(args[0]) : 7;

    final Fact = List<int>.filled(n+1,0);    
    Fact[0] = 1;
    for (int i = 1; i < Fact.length; i++) {
        Fact[i] = Fact[i-1] * i;
    }     
       
    const NCHUNKS = 4; 
    int CHUNKSZ = (Fact[n] + NCHUNKS - 1) ~/ NCHUNKS;
    CHUNKSZ += CHUNKSZ % 2; 
    
    int maxflips = 0, checksum = 0;        
    ReceivePort _replyPort = new ReceivePort();          
    int awaited = NCHUNKS;          
           
    _replyPort.listen((dynamic stuff) {
        awaited--;
        final reply = stuff as Reply;
        checksum += reply.sum; 
        if (reply.flips > maxflips){
            maxflips = reply.flips;       
        }        
              
        if (awaited < 1){
           print("$checksum\nPfannkuchen($n) = $maxflips");                   
            _replyPort.close();          
        }         
    });     
        
    Isolate.spawn(fannkuch,Request(n,0,CHUNKSZ,Fact,_replyPort.sendPort));        
}

class Request {
    int n = 0, min = 0, CHUNKSZ = 0;    
    List<int> Fact;     
    SendPort ch;   
    Request(this.n, this.min, this.CHUNKSZ, this.Fact, this.ch);
}

class Reply {
    int sum = 0, flips = 0;
    Reply(this.sum, this.flips);
}
