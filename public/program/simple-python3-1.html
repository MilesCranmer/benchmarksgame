# The Computer Language Benchmarks Game
# https://salsa.debian.org/benchmarksgame-team/benchmarksgame/

# contributed by Brent Fulgham

import sys

def main():
    cout = sys.stdout.write
    bit_num = 0
    byte_acc = 0
    iter = 50
    limit2 = 4.
    w = h = int(sys.argv[1])

    cout("P4\n%d %d\n" % (w, h))

    for y in range(h):
        fy = 2j * y / h - 1j
        for x in range(w):
            z = 0j
            c = 2. * x / w - 1.5 + fy

            byte_acc <<= 1
            bit_num += 1

            for i in range(iter):
                z = z * z + c
                if z.real * z.real + z.imag * z.imag > limit2:
                    break
            else:
                byte_acc += 1

            if bit_num == 8:
                cout(chr(byte_acc))
                byte_acc = bit_num = 0
            elif x == w - 1:
                byte_acc = byte_acc << (8 - w % 8)
                cout(chr(byte_acc))
                byte_acc = bit_num = 0

main()
