const revcompdata = Dict(
   'A'=> 'T', 'a'=> 'T',
   'C'=> 'G', 'c'=> 'G',

   'S'=> 'S', 's'=> 'S',
   'Y'=> 'R', 'y'=> 'R',
   'K'=> 'M', 'k'=> 'M',
   'V'=> 'B', 'v'=> 'B',
   'H'=> 'D', 'h'=> 'D',
   'D'=> 'H', 'd'=> 'H',
   'B'=> 'V', 'b'=> 'V',
   'N'=> 'N', 'n'=> 'N',
)

revcomp = (s)-> map(c -> revcompdata[c] , s)

function print_result(header, data)
   r_data= revcomp(reverse(data))
   println(header)
   l = length(r_data)
   for i in 1:60:l
      println(r_data[i:min(i+59, l)])
   end
end

function main()
   data = ""
   header = ""
   while !isempty(@eval line = readline())
      if line[1] == '>'
         if !isempty(data)
            print_result(header, data)
            data = ""
         end
         header = line
      else
         data *= line
      end
   end
   if !isempty(data)
      print_result(header, data)
   end
end
