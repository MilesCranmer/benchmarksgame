    using System;

    using V256d = System.Runtime.Intrinsics.Vector256<double>;

    public static unsafe class NBody {
        [StructLayout(LayoutKind.Explicit, Pack = 8, Size = 64)]
        struct Body {
            [FieldOffset(sizeof(double) * 0)] public double x;
            [FieldOffset(sizeof(double) * 1)] public double y;
            [FieldOffset(sizeof(double) * 2)] public double z;
            [FieldOffset(sizeof(double) * 3)] public double __filler__;

            [FieldOffset(sizeof(double) * 4)] public double vx;
            [FieldOffset(sizeof(double) * 5)] public double vy;
            [FieldOffset(sizeof(double) * 6)] public double vz;
            [FieldOffset(sizeof(double) * 7)] public double mass;

            [FieldOffset(sizeof(double) * 0)] public V256d Position;
            [FieldOffset(sizeof(double) * 4)] public V256d Velocity;
        }

        static void PrintEnergyAVX(int length, Body* system, V256d* masses) {
            Console.Out.WriteLine(EnergyAVX(length, system, masses).ToString("F9"));
        }

        // Moved out of main because I don't want to apply AggressiveOptimization
        // to code that isn't hot. Doing so will increase startup time

        [MethodImpl(AggressiveOptimization)]
        static void RunSystemAVX( int iterations, 
                                                            int length, 
                                                            double stepSize, 
                                                            Body* system, 
                                                            V256d* masses) {
            unchecked {
                V256d stepRes = Vector256.Create(stepSize, stepSize, stepSize, 0d);
                while (iterations-- > 0) {
                    AdvanceAVX(length, stepRes, system, masses);
                }
            }
        }

        [MethodImpl(AggressiveInlining | AggressiveOptimization)]
        static void AdvanceAVX( int length, 
                                                        V256d stepRes, 
                                                        Body* system, 
                                                        V256d* masses) {
            // I tried doing "reciprocal square root approximation" as shown
            //    in the C++ #7 but it actually made it 4 times slower
            unchecked {
                for (int i = 0; i < length; ++i) {
                    V256d iPosition = (system + i)->Position;
                    V256d iVelocity = (system + i)->Velocity;
                    V256d iMass = *(masses + i);
                    for (int j = i + 1; j < length; ++j) {
                        V256d dx = Avx.Subtract((system + j)->Position, iPosition);
                        double dp = dx.DotProduct();
                        double mag = stepRes.ToScalar() / (dp * Math.Sqrt(dp));
                        V256d mags = Avx.Multiply(dx, 
                                Avx.BroadcastScalarToVector256(&mag));                       
                        (system + j)->Velocity = Avx.Subtract((system + j)->Velocity, 
                                Avx.Multiply(iMass, mags));
                        iVelocity = Avx.Add(iVelocity, 
                                Avx.Multiply(*(masses + j), mags));
                    } // End of j loop
                    (system + i)->Position = Avx.Add(iPosition, 
                            Avx.Multiply(iVelocity, stepRes));
                    (system + i)->Velocity = iVelocity;
                } // End of i loop
            }
        }

        private static void OffsetMomentumAVX(int length, Body* system, V256d* masses) {
            unchecked {
                V256d velocity = system->Velocity;
                for (var i = 1; i < length; ++i) {
                    velocity = Avx.Add(velocity, 
                            Avx.Multiply((system + i)->Velocity, masses[i]));
                }
                system->Velocity = Avx.Divide(velocity, Avx.Multiply(
                        Vector256.Create(-1d), *masses) )
                                .WithElement(3, (*masses).ToScalar());
            }
        }

        // NOTE: This is actually slower than the scaler version in Derek_AVX_V3
        // But the goal is to use Avx where possible. 
        
        static double EnergyAVX(int length, Body* system, V256d* masses) {
            unchecked {
                double e = 0.0;
                for (int i = 0; i < length; ++i) {
                    double iMass = (masses + i)->ToScalar();
                    V256d iPos = (system + i)->Position;
                    e += 0.5 * ((system + i)->Velocity)
                            .WithElement(3, 0d).DotProduct() * iMass;
                    for (int j = i + 1; j < length; ++j) {
                        e -= iMass * (masses + j)->ToScalar()  / Math.Sqrt(
                                Avx.Subtract( iPos, (system + j)->Position)
                                        .WithElement(3, 0d).DotProduct());
                    }
                }
                return e;
            }
        }

        [MethodImpl(AggressiveInlining | AggressiveOptimization)]
        unsafe static double DotProduct(this V256d dx) {
            V256d mult = Avx.Multiply(dx, dx);
            V256d tmp = Avx.HorizontalAdd(mult, mult);
            return Sse2.Add(tmp.GetLower(), tmp.GetUpper()).ToScalar();
        }
    }
