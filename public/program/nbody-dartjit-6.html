import 'dart:math' as Math;

const SOLAR_MASS = 4 * Math.pi * Math.pi;
const DAYS_PER_YEAR = 365.24;

class Body {
  double x = 0, y = 0, z = 0, vx = 0, vy = 0, vz = 0, mass = 0;
  Body(this.x, this.y, this.z, this.vx, this.vy, this.vz, this.mass);
}

final bodies = [Sun(), Jupiter(), Saturn(), Uranus(), Neptune()];

void offsetMomentum() {
  var px = 0.0;
  var py = 0.0;
  var pz = 0.0;
  final size = bodies.length;
  for (var i = 0; i < size; i++) {
    final body = bodies[i];
    final mass = body.mass;
    px += body.vx * mass;
    py += body.vy * mass;
    pz += body.vz * mass;
  }

  final body = bodies[0];
  body.vx = -px / SOLAR_MASS;
  body.vy = -py / SOLAR_MASS;
  body.vz = -pz / SOLAR_MASS;
}

void advance(double dt) {
  final size = bodies.length;

  for (var i = 0; i < size; i++) {
    final bodyi = bodies[i];
    var vxi = bodyi.vx;
    var vyi = bodyi.vy;
    var vzi = bodyi.vz;
    for (var j = i + 1; j < size; j++) {
      final bodyj = bodies[j];
      final dx = bodyi.x - bodyj.x;
      final dy = bodyi.y - bodyj.y;
      final dz = bodyi.z - bodyj.z;

      final d2 = dx * dx + dy * dy + dz * dz;
      final mag = dt / (d2 * Math.sqrt(d2));

      final massj = bodyj.mass;
      vxi -= dx * massj * mag;
      vyi -= dy * massj * mag;
      vzi -= dz * massj * mag;

      final massi = bodyi.mass;
      bodyj.vx += dx * massi * mag;
      bodyj.vy += dy * massi * mag;
      bodyj.vz += dz * massi * mag;
    }
    bodyi.vx = vxi;
    bodyi.vy = vyi;
    bodyi.vz = vzi;
  }

  for (var i = 0; i < size; i++) {
    final body = bodies[i];
    body.x += dt * body.vx;
    body.y += dt * body.vy;
    body.z += dt * body.vz;
  }
}

double energy() {
  var e = 0.0;
  final size = bodies.length;

  for (var i = 0; i < size; i++) {
    final bodyi = bodies[i];

    e += 0.5 *
        bodyi.mass *
        (bodyi.vx * bodyi.vx + bodyi.vy * bodyi.vy + bodyi.vz * bodyi.vz);

    for (var j = i + 1; j < size; j++) {
      final bodyj = bodies[j];
      final dx = bodyi.x - bodyj.x;
      final dy = bodyi.y - bodyj.y;
      final dz = bodyi.z - bodyj.z;

      final distance = Math.sqrt(dx * dx + dy * dy + dz * dz);
      e -= (bodyi.mass * bodyj.mass) / distance;
    }
  }
  return e;
}

void main(List<String> args) {
  final n = (args.length > 0) ? int.parse(args[0]) : 1000;

  offsetMomentum();
  print(energy().toStringAsFixed(9));
  for (int i = 0; i < n; i++) {
    advance(0.01);
  }
  print(energy().toStringAsFixed(9));
}
