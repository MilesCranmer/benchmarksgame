// The Computer Language Benchmarks Game
// https://salsa.debian.org/benchmarksgame-team/benchmarksgame/

// Contributed by Andrei Simion

#include <cstdio>
#include <cstdlib>
#include <array>
#include <cmath>

constexpr double pi = 3.141592653589793;
constexpr double solar_mass = 4 * pi * pi;
constexpr double days_per_year = 365.24;
constexpr int num_bodies = 5;
constexpr int interactions = (num_bodies - 1) * num_bodies / 2;

struct Body
{
    double x, y, z, vx, vy, vz, mass;

    double get_velocity_square_sum()
    {
        return vx * vx + vy * vy + vz * vz;
    }
};


class NBodySystem
{
public:

    double compute_energy()
    {
        double energy = 0.0;
        for (std::size_t i = 0; i < bodies.size(); ++i)
        {
            energy += 0.5 * bodies[i].mass * bodies[i].get_velocity_square_sum();
  
            for (std::size_t j = i + 1; j < bodies.size(); ++j)
            {
                double dx = bodies[i].x - bodies[j].x;
                double dy = bodies[i].y - bodies[j].y;
                double dz = bodies[i].z - bodies[j].z;
                double distance = std::sqrt(dx * dx + dy * dy + dz * dz);

                energy-= (bodies[i].mass * bodies[j].mass) / distance;
            }
        }
        return energy;
    }
   
   void advance(const std::size_t n)
   {
       constexpr double dt = 0.01;
      
       double dxs[interactions], dys[interactions], dzs[interactions];
       double magnitudes[interactions];

       for (std::size_t  i = 0; i < n; ++i)
       {
           for (std::size_t i = 0, k = 0; i < bodies.size(); ++i)
           {
               for (std::size_t j = i + 1; j < bodies.size(); ++j, ++k)
               {
                   dxs[k] = bodies[i].x - bodies[j].x;
                   dys[k] = bodies[i].y - bodies[j].y;
                   dzs[k] = bodies[i].z - bodies[j].z;

                   const double d_squared = dxs[k] * dxs[k] + dys[k] * dys[k] + dzs[k] * dzs[k];
                   const double distance = std::sqrt(d_squared);
				
                   magnitudes[k] = dt / (d_squared * distance);
                }
            }

            for (std::size_t i = 0, k = 0; i < bodies.size(); ++i)
            {
                for (std::size_t j = i + 1; j < bodies.size(); ++j, ++k)
                {
                    const double mm1 = bodies[j].mass * magnitudes[k];
                    const double mm2 = bodies[i].mass * magnitudes[k];
 
                    bodies[i].vx -= dxs[k] * mm1;
                    bodies[i].vy -= dys[k] * mm1;					
                    bodies[i].vz -= dzs[k] * mm1;
 
                    bodies[j].vx += dxs[k] * mm2;
                    bodies[j].vy += dys[k] * mm2;
                    bodies[j].vz += dzs[k] * mm2;
                }
        
#ifndef __clang__
                // g++ gives 1-2% better performance 
                // if the update happens here
                bodies[i].x += dt * bodies[i].vx;
                bodies[i].y += dt * bodies[i].vy;
                bodies[i].z += dt * bodies[i].vz;
#endif
            }
            
#ifdef __clang__
            for (std::size_t i = 0; i < bodies.size(); ++i)
            {
                bodies[i].x += dt * bodies[i].vx;
                bodies[i].y += dt * bodies[i].vy;
                bodies[i].z += dt * bodies[i].vz;
            }
#endif
       }
    }

private:
    std::array<Body, num_bodies> bodies = create_bodies();
 
    constexpr static double compute_p(double Body::* member,
                            const std::array<Body, num_bodies - 1>& bodies)
    {
        double p = 0.0;
        for (size_t i = 0; i < bodies.size(); ++i)
        {
            p += bodies[i].*member * bodies[i].mass;
        }
        return p;
    }

    constexpr static std::array<Body, num_bodies> create_bodies()
    {
        constexpr Body jupiter = { .x    = 4.84143144246472090e+00,
                                   .y    = -1.16032004402742839e+00,
                                   .z    = -1.03622044471123109e-01,
                                   .vx   = 1.66007664274403694e-03 * days_per_year,
                                   .vy   = 7.69901118419740425e-03 * days_per_year,
                                   .vz   = -6.90460016972063023e-05 * days_per_year,
                                   .mass = 9.54791938424326609e-04 * solar_mass };

        constexpr Body saturn  = { .x    = 8.34336671824457987e+00,
                                   .y    = 4.12479856412430479e+00,
                                   .z    = -4.03523417114321381e-01,
                                   .vx   = -2.76742510726862411e-03 * days_per_year,
                                   .vy   = 4.99852801234917238e-03 * days_per_year,
                                   .vz   = 2.30417297573763929e-05 * days_per_year,
                                   .mass = 2.85885980666130812e-04 * solar_mass };

        constexpr Body uranus  = { .x    = 1.28943695621391310e+01,
                                   .y    = -1.51111514016986312e+01,
                                   .z    = -2.23307578892655734e-01,
                                   .vx   = 2.96460137564761618e-03 * days_per_year,
                                   .vy   = 2.37847173959480950e-03 * days_per_year,
                                   .vz   = -2.96589568540237556e-05 * days_per_year,
                                   .mass = 4.36624404335156298e-05 * solar_mass };

        constexpr Body neptune = { .x    = 1.53796971148509165e+01,
                                   .y    = -2.59193146099879641e+01,
                                   .z    = 1.79258772950371181e-01,
                                   .vx   = 2.68067772490389322e-03 * days_per_year,
                                   .vy   = 1.62824170038242295e-03 * days_per_year,
                                   .vz   = -9.51592254519715870e-05 * days_per_year,
                                   .mass = 5.15138902046611451e-05 * solar_mass };

        constexpr std::array<Body, num_bodies - 1> aux_bodies = { jupiter, saturn, uranus, neptune };

        constexpr double px = compute_p(&Body::vx, aux_bodies);
        constexpr double py = compute_p(&Body::vy, aux_bodies);
        constexpr double pz = compute_p(&Body::vz, aux_bodies);
        
        constexpr Body sun = { .x  = 0,
                               .y  = 0,
                               .z  = 0,
                               .vx = -px / solar_mass,
                               .vy = -py / solar_mass,
                               .vz = -pz / solar_mass,
                               .mass = solar_mass };

        return { sun, jupiter, saturn, uranus, neptune };
    }   
};


int main(int argc, char** argv)
{
    std::size_t n = atoi(argv[1]);

    NBodySystem system;

    printf("%.9f\n", system.compute_energy());

    system.advance(n);

    printf("%.9f\n", system.compute_energy());
}
