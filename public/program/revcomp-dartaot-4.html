import 'dart:async';
import 'dart:io';

const begSequence = 62; //">"
const endOfLine = 10; //"\n"
const Map<String, String> rules = {
  'A': 'T',
  'B': 'V',
  'C': 'G',
  'D': 'H',
  'G': 'C',
  'H': 'D',
  'K': 'M',
  'M': 'K',
  'N': 'N',
  'R': 'Y',
  'S': 'S',
  'T': 'A',
  'U': 'A',
  'V': 'B',
  'Y': 'R',
  'W': 'W',
};

final List<int> complement = initComplement();

List<int> initComplement() {
  var res = List<int>(256);
  for (var i = 0; i < 255; i++) {
    var c = String.fromCharCode(i).toUpperCase();
    if (rules[c] != null) {
      res[i] = rules[c].codeUnits[0];
    } else {
      res[i] = i;
    }
  }
  return res;
}

List<int> flush(List<int> seq, int hs, int he, int end) {
  //reverse and complement
  for (int i = he + 1, j = end - 1; i < j; i++, j--) {
    while (seq[i] == endOfLine) i++;
    if (i > j) break;
    while (seq[j] == endOfLine) j--;
    if (i > j) break;
    int si = seq[i];
    seq[i] = complement[seq[j]];
    seq[j] = complement[si];
  }
  //output
  return seq.getRange(hs, end).toList();
}

Stream<List<int>> run(Stream<List<int>> input) async* {
  var inDNA = true;
  List<int> seq = [];
  int hs;
  int he;
  int i = 0;
  await for (final buf in input) {
    seq.addAll(buf);
    for (var j = 0; j < buf.length; j++, i++) {
      if (inDNA) {
        if (buf[j] == begSequence) {
          inDNA = false;
          if (he != null) {
            yield flush(seq, hs, he, i);
						seq = buf.getRange(j, buf.length).toList();
            i = 0;
            he = null;
          }
          hs = i;
        }
      } else if (buf[j] == endOfLine) {
        inDNA = true;
        he = i;
      }
    }
  }
  if (he != null) yield flush(seq, hs, he, i);
}
