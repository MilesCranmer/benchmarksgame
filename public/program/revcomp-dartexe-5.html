import 'dart:io';
import 'dart:isolate';
import 'dart:typed_data';

final nl = '\n'.codeUnitAt(0);
final def = '>'.codeUnitAt(0); // Definition-line.

void main() async {
  final mainIsolate = ReceivePort();
  final waiting = <SendPort>[]; // Grows without end!
  var finalCount = 0;
  var nextToWrite = 0;

  mainIsolate.listen((dynamic p) {
    if (p is SendPort) {
      waiting.add(p);
      p.send(nextToWrite);
    } else {
      if (++nextToWrite == finalCount) {
        mainIsolate.close();
      } else {
        for (var w in waiting) {
          w.send(nextToWrite);
        }
      }
    }
  });

  var blocks = <Uint8List>[];
  int k = 0;
  var sequenceNumber = 0;
  var prev = Uint8List(0), bytes = Uint8List(0);

  void separateSequences() {
    if (blocks.isNotEmpty) {
      if (bytes == blocks.first) {
        if (blocks.length > 1) {
          blocks.removeLast();
        }
      } else if (bytes != blocks.last) {
        blocks.add(bytes.sublist(0, k)); // Last of current sequence.
        prev = bytes.sublist(k, bytes.length); // First of next sequence.
      }
      Isolate.spawn(
          other, Sequence(mainIsolate.sendPort, blocks, sequenceNumber++));
    }
    if (prev.isNotEmpty) {
      blocks = [prev];
      prev = Uint8List(0);
      k += bytes.length; // Skip to next block.
    } else {
      blocks = [];
    }
  }

  await for (var data in stdin) {
    bytes = data as Uint8List;
    k = 0;
    // Scan within 64x1024 block.
    while (k < bytes.length) {
      k = bytes.indexOf(def, k);
      if (k == -1) {
        // Next definition-line not found.
        blocks.add(bytes);
        k += bytes.length; // Skip to next block.
      } else {
        // Found next definition-line.
        separateSequences();
      }
      k++;
    }
  }
  separateSequences();
  finalCount = sequenceNumber;
}

void other(Sequence seq) {
  var builder = BytesBuilder(copy: false);
  for (var k = 0; k < seq.blocks.length; k++) {
    builder.add(seq.blocks[k]);
  }
  var buf = builder.takeBytes();

  // Walk from both ends of the sequence, complementing and swapping.
  final lut = codeComplements();
  var lo = buf.indexOf(nl) + 1;
  var hi = buf.length - 1;
  var loCode = 0, hiCode = 0;

  while (lo < hi) {
    if (buf[lo] == nl) { // Skip.
      if (++lo < hi) {
        loCode = buf[lo];
      } else {
        break;
      }
    } else {
      loCode = buf[lo];
    }
    if (buf[hi] == nl) { // Skip.
      if (--hi > lo) {
        hiCode = buf[hi];
      } else {
        break;
      }
    } else {
      hiCode = buf[hi];
    }
    buf[lo++] = lut[hiCode];
    buf[hi--] = lut[loCode];
  }
  
  final otherIsolate = ReceivePort();
  otherIsolate.listen((dynamic nextToWrite) {
    if (nextToWrite == seq.number) {
      stdout.write(String.fromCharCodes(buf));      
      seq.p.send(1); // Sequence written — increment nextToWrite.
      seq.blocks.clear();
    }
  });
  // Ready to write, so ask which sequence is next to write.
  seq.p.send(otherIsolate.sendPort);  
}

class Sequence {
  SendPort p;
  List<Uint8List> blocks;
  int number = 0;
  Sequence(this.p, this.blocks, this.number);
}

Uint8List codeComplements() {
  // Based on Brad Chamberlain / Ben Harshbarger Chapel program.
  var pairs = 'ATCGGCTAUAMKRYWWSSYRKMVBHDDHBVNN'.codeUnits;
  var upperToLower = 'a'.codeUnitAt(0) - 'A'.codeUnitAt(0);
  var table = Uint8List(128);
  for (var i = 0; i < pairs.length; i += 2) {
    final src = pairs[i], dst = pairs[i + 1];
    table[src] = dst;
    table[src + upperToLower] = dst;
  }
  return table;
}
