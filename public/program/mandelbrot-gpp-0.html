#include <iostream>
#include <vectorclass.h>
#include <array>
#include <vector>
#include <thread>

// Number of vectors concurrently computed in the inner loop
#define N 3
// Nb of threads
#define NBTHREADS 4

/**
 * This defines an array of items of a given VCL vector type
 * and implements a number of operators and useful method to
 * handle it as if we were using a single vector element
 */
template<typename VectorType, int NbElem = N>
  struct Array : std::array<VectorType, NbElem> {
  using BaseType = decltype(std::declval<VectorType>().extract(0));
  using BoolType = decltype(std::declval<VectorType>() > std::declval<BaseType>());
  using BoolArrayType = Array<BoolType, NbElem>;
  Array() = default;
  Array(BaseType v) {
    for (int i = 0; i < NbElem; i++) (*this)[i] = v;
  }
  Array(BaseType minx, BaseType dx) {
    for (int i = 0; i < NbElem; i++) {
      (*this)[i] = {minx+(7+i*8)*dx, minx+(6+i*8)*dx, minx+(5+i*8)*dx, minx+(4+i*8)*dx, minx+(3+i*8)*dx, minx+(2+i*8)*dx, minx+(1+i*8)*dx, minx+i*8*dx};
    }
  }
  void store(BaseType* buffer) {
    for (int i = 0; i < NbElem; i++) {  
      (*this)[i].store(buffer+8*i);
    }
  }
  Array operator+=(Array& arg) {
    for (int i = 0; i < NbElem; i++) (*this)[i] += arg;
  }
  Array operator+=(BaseType arg) {
    for (int i = 0; i < NbElem; i++) (*this)[i] += arg;
    return *this;
  }
  Array operator+(Array const & arg) {
    Array res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] + arg[i];
    return res;
  }
  Array operator+(VectorType arg) {
    Array res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] + arg;
    return res;
  }
  Array operator-(Array const& arg) {
    Array res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] - arg[i];
    return res;
  }
  Array operator*(Array const & arg) {
    Array res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] * arg[i];
    return res;
  }
  Array operator*(BaseType arg) {
    Array res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] * arg;
    return res;
  }
  BoolArrayType operator<(BaseType arg) {
    BoolArrayType res;
    for (int i = 0; i < NbElem; i++) res[i] = (*this)[i] < arg;
    return res;
  }
};

/**
 * SPecialization of Array for boolean types where arithmetic
 * operation do not make sense but horizontal_or is useful
 */
template<int NbElem>
  struct Array<Vec8fb, NbElem> : std::array<Vec8fb, NbElem> {
  using BaseType = decltype(std::declval<Vec8fb>().extract(0));
  Array() = default;
  Array(BaseType v) {
    for (int i = 0; i < NbElem; i++) (*this)[i] = v;
  }
  Array& operator|=(Array arg) {
    for (int i = 0; i < NbElem; i++) (*this)[i] |= arg[i];
    return *this;
  }
  bool horizontal_or() {
    for (int i = 0; i < N; i++) if (::horizontal_or((*this)[i])) return true;
    return false;
  }

};

/**
 * kernel of mandelbrot compilation compiling number
 * of iterations for a given set of pixels
 */
void kernel(char* buf, Array<Vec8f> ax, Vec8f ay) {
  // future results, all to -1 initially
  Array<Vec8ib> res{false};
  // remember which pixels are already diverging
  Array<Vec8fb> cmp{false};

  Array<Vec8f> x{0};
  Array<Vec8f> y{0};
  for (int n = 1; n <= 50; n++) {
    auto newx = x*x - y*y + ax;
    auto newy = x*y*2 + ay;
    cmp = (newx*newx + newy*newy < 4);
    // no need to continue if all pixels are over
    if (!cmp.horizontal_or()) {
      break;
    }
    x = newx;
    y = newy;
  }
  for (int i = 0; i < N; i++) buf[i] = to_bits(cmp[i]);
}

/**
 * fills a char buffer with mandelbrot values for each
 * pixel of an image
 */
void mandel(char* buffer,
            const float minx,
            const float dx,
            const unsigned int nx,
            const float miny,
            const float dy,
            const unsigned int ny) {
  Vec8f ay{miny}; // vector of 8 x miny value
  // loop one pixel at a time in y
  for (unsigned int any = 0; any < ny; ay += dy, any++) {
    // first N vectors of 8 pixels in x
    Array<Vec8f> ax{minx, dx};
    // loop 8*N by 8*N pixels in x
    for (unsigned int anx = 0; anx < nx; anx += N*8) {
      // compute and store back 8 pixels in one go
      //std::cout << "offset " << anx << " " << (any*nx+anx)/8 << std::endl;
      kernel(buffer+(any*nx+anx)/8, ax, ay);
      ax += N*8*dx;
    }
  }
}

/**
 * main method :
 *  - parses input for size of the picture
 *  - start threads to compute values in parallel
 *  - write output file to stdout
 */
int main(int argc, char** argv) { 
  std::ios::sync_with_stdio(false);
  unsigned int n = 16000;
  if (argc>=2) {
    n = atoi(argv[1]);
  }
  constexpr float minx = -1.5;
  constexpr float maxx = 0.5;
  constexpr float miny = -1.0;
  constexpr float maxy = 1.0;
  auto dx = (maxx - minx) / n;
  auto dy = (maxy - miny) / n;
  auto nyThreads = (n+NBTHREADS-1)/NBTHREADS;
  auto dyThreads = nyThreads*dy;
  auto dData = n*nyThreads/8;
  // make sure we have space for full vectors of vectors
  // as last one may compute and write extra data after the end
  auto nrounded = (n+8*N-1)/8/N*8*N;
  auto dDatarounded = nrounded*nyThreads/8;
  std::vector<char> buf;
  buf.resize(dDatarounded*NBTHREADS);
  //std::cout << nyThreads << " " << dData << " " << buf.size() << "\n";
  std::vector<std::thread> threads;
  for (unsigned int t = 0; t < NBTHREADS; t++) {
    threads.emplace_back(mandel, buf.data()+t*dDatarounded,
                         minx, dx, n,
                         miny+t*dyThreads, dy, nyThreads);
  }
  for (auto& t : threads) {
    t.join();
  }
  std::cout << "P4\n" << n << " " << n << "\n";
  for (unsigned int t = 0; t < NBTHREADS; t++) {
    auto bufn = buf.data()+t*dDatarounded;
    std::cout.write(bufn, dData);
  }
}
